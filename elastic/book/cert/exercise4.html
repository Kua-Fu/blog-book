二、自定义分词

2.1 题目

正确的排除 ' 对查询的影响, 确保对 waynes/wayne's, kings/king's 的查询有一样的结果和分值 (custom analyzer)

2.2 分析

自定义分词器，char filters 把 ' 过滤


PUT yztest-4-1
{
  "settings": {
     "analysis": {
      "analyzer": {
        "my_custom_analyzer": {
          "type": "custom", 
          "tokenizer": "standard",
          "char_filter": [
            "my_char_filter"
          ]
        }
      },
      "char_filter": {
        "my_char_filter": {
          "type": "mapping",
          "mappings": [
            "' => "
          ]
        }
      }
    }
  },
  "mappings": {
    "properties": {
      "f1": {
        "type": "text",
        "analyzer": "my_custom_analyzer"
      }
    }
  }
}

GET yztest-4-1/_analyze
{
  "field": "f1", 
  "text": "tom's toms"
}

POST yztest-4-1/_bulk
{"index":{}}
{"f1": "tom's"}
{"index":{}}
{"f1": "toms"}

GET yztest-4-1/_search
{
  "query": {
    "match": {
      "f1": "tom's"
    }
  }
}
